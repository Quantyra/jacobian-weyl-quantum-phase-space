\begin{abstract}
We isolate a structural lemma for dual vector fields of local diffeomorphisms
\(F:\mathbb{R}^n\to\mathbb{R}^n\): if global dual fields \(X_j\) with \(DF\cdot X_j=e_j\)
are all complete, then \(F\) is surjective. Hence any non-surjective dual system
has at least one incomplete dual field. We propose a classification schema
(B-incomplete / B-ESS / B-Def / B-CCR / B-CP) and populate it on atlas rows
A000--A006 (pass and fail baselines, Keller seed A001, family pilot A002).
No gates, channels, or computational advantage are claimed. Companion pack:
\url{https://github.com/Quantyra/jacobian-weyl-quantum-phase-space/blob/main/docs/notes/COMPANION-PACK.md}.
\end{abstract}

\section{Structural lemma}
\begin{lemma}[Complete dual fields imply surjectivity]
\label{lem:B1}
Let \(F:\mathbb{R}^n\to\mathbb{R}^n\) be \(C^\infty\) and a local diffeomorphism
with dual fields \(X_j\), \(DF\cdot X_j=e_j\). If every \(X_j\) is complete, then
\(F(\mathbb{R}^n)=\mathbb{R}^n\).
\end{lemma}
\begin{proof}
\(F(\phi_t^{(j)}(q))=F(q)+te_j\); iterate; image open and translation-invariant.
\end{proof}
\begin{corollary}
Non-surjective dual systems have some incomplete \(X_j\).
\end{corollary}

\section{Examples (new in v0.2)}
\begin{example}[A005 --- linear stretch, pass-like]
\(F=\mathrm{diag}(2,\tfrac12,1)\), \(\det DF=1\). Dual fields
\(X_0=\tfrac12\partial_x\), \(X_1=2\partial_y\), \(X_2=\partial_z\) are complete.
CAS: \texttt{cas\_atlas\_A005\_A006\_B001.json}.
\end{example}

\begin{example}[A006 --- exp quadrant, fail-like]
\(F(x,y,z)=(e^x,e^y,z)\), \(\det DF=e^{x+y}>0\), image
\((0,\infty)^2\times\mathbb{R}\). Dual fields \(X_0=e^{-x}\partial_x\),
\(X_1=e^{-y}\partial_y\) incomplete backward (\(e^x=e^{x_0}+t\)). Matches
Lemma~\ref{lem:B1}. CAS same file.
\end{example}
