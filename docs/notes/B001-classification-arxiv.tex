\begin{abstract}
We isolate a structural lemma for dual vector fields of local diffeomorphisms
\(F:\mathbb{R}^n\to\mathbb{R}^n\): if global dual fields \(X_j\) with \(DF\cdot X_j=e_j\)
are all complete, then \(F\) is surjective. Hence any non-surjective dual system
has at least one incomplete dual field. We propose a classification schema
(B-incomplete / B-ESS / B-Def / B-CCR / B-CP) and populate it on atlas rows
A000--A010 (pass and fail baselines, Keller seed A001, family pilot A002,
a smooth 2D pair A007/A008 illustrating that the forced incomplete set can
be as thin as measure zero, and a polynomial 2D pair A009/A010: Pinchuk's
real-plane map with \emph{open-set} incompleteness, and a polynomial shear
automorphism with complete dual fields). We prove restricted dichotomy lemmas:
polynomial product local diffeos have complete dual fields; the A007
zero-dimensional asymptotic mechanism is impossible for polynomials (Jelonek);
in dimension~2 dual incompleteness is fiber-saturated and falls into empty /
open / residual-thin regimes. Residual-thin regime~T is excluded on proper,
injective, degree-one-component, product, and triangular classes, and reduced
to incompleteness only on atypical fibers; full exclusion of~T remains open.
No gates, channels, or computational advantage are claimed. Companion pack:
\url{https://github.com/Quantyra/jacobian-weyl-quantum-phase-space/blob/main/docs/notes/COMPANION-PACK.md}.
\end{abstract}

\section{Structural lemma}
\begin{lemma}[Complete dual fields imply surjectivity]
\label{lem:B1}
Let \(F:\mathbb{R}^n\to\mathbb{R}^n\) be \(C^\infty\) and a local diffeomorphism
with dual fields \(X_j\), \(DF\cdot X_j=e_j\). If every \(X_j\) is complete, then
\(F(\mathbb{R}^n)=\mathbb{R}^n\).
\end{lemma}
\begin{proof}
\(F(\phi_t^{(j)}(q))=F(q)+te_j\); iterate; image open and translation-invariant.
\end{proof}
\begin{corollary}
Non-surjective dual systems have some incomplete \(X_j\).
\end{corollary}

\section{Examples (new in v0.3): two incompleteness regimes}
\begin{example}[A007 --- complex exponential, thin fail-like]
\(F(x,y,z)=(e^x\cos y,\,e^x\sin y,\,z)\), i.e.\ \(z\mapsto e^z\) padded.
Conformal, \(\det DF=e^{2x}>0\); image misses the origin (non-surjective).
Writing \(w=e^{x+iy}\), the dual field \(X_0\) solves \(\dot w=1\), so
\(w(t)=w(0)+t\). If \(\operatorname{Im}w(0)\neq 0\) (full measure), the orbit
never reaches \(w=0\): \textbf{complete}. If \(\operatorname{Im}w(0)=0\) (a
measure-zero pair of rays), \(w(t)\) hits \(0\) at finite \(t\) and
\(x(t)\to-\infty\): \textbf{incomplete}. Thus Lemma~\ref{lem:B1}'s corollary
is satisfied by an incomplete set of \emph{measure zero}, sharply contrasting
A006/A001 where incompleteness occupies an \emph{open} set. CAS:
\texttt{cas\_atlas\_A007\_A008\_B001.json}.
\end{example}

\begin{example}[A008 --- shear + sine, 2D global diffeomorphism]
\(F(x,y,z)=(x,\,y+\sin x,\,z)\), \(\det DF=1\). Explicit global inverse
\((x,y)\mapsto(x,y-\sin x)\) (verified by CAS roundtrip). Dual fields
\(X_0=\partial_x-\cos x\,\partial_y\), \(X_1=\partial_y\); the \(X_0\) flow
\(y(t)=y_0-\sin(x_0+t)+\sin(x_0)\) is bounded for all \(t\): \textbf{complete}.
A genuinely nonlinear, non-triangular-affine 2D coupling (unlike A004/A005)
that nonetheless has no incompleteness at all.
\end{example}

\section{Examples (new in v0.4): polynomial 2D evidence}
\begin{example}[A009 --- Pinchuk real plane map, open fail-like]
\label{ex:A009}
Let \(t=xy-1\), \(h=t(xt+1)\), \(f=(xt+1)^2(t^2+y)\), and
\[
P=f+h,\qquad
Q=-t^2-6th(h+1)-u(f,h),
\]
with the Campbell/van~den~Essen auxiliary
\(u=170fh+91h^2+195fh^2+69h^3+75fh^3+\tfrac{75}{4}h^4\)
(Pinchuk~\cite{Pin94}, Campbell~\cite{Cam10}). Then
\(F=(P,Q):\mathbb{R}^2\to\mathbb{R}^2\) (padded by idle \(z\)) is polynomial of
degrees \((10,25)\), with
\[
\det DF=t^2+\bigl(t+f(13+15h)\bigr)^2+f^2>0
\]
everywhere (sum of squares; \(t=f=0\) is impossible over \(\mathbb{R}\)).
It is not injective (CAS-certified collisions) and not surjective (misses
exactly two points~\cite{Cam10}). The dual field
\(X_1=H(P)/j=(-P_y,P_x)/j\) satisfies \(DF\cdot X_1=e_1\) and \(\dot Q=1\).
On every level \(P=c\) with \(c>-1\), the real level curve has poles
\(h=-1\pm\sqrt{1+c}\) at which \((x,y)\to\infty\) while \(Q\) approaches a
finite asymptotic value: escape in finite \(X_1\)-time. Hence the
incompleteness locus of \(X_1\) contains the open set \(\{P>-1\}\).
\textbf{Open-set incompleteness for a genuinely polynomial 2D Keller-type map},
contrasting A007's thin failure. CAS:
\texttt{cas\_atlas\_A009\_A010\_B001.json}.
\end{example}

\begin{example}[A010 --- polynomial shear automorphism, pass-like]
\(F(x,y,z)=(x,\,y+x^2,\,z)\), \(\det DF=1\), inverse \((x,y-x^2,z)\).
Dual fields \(X_0=\partial_x-2x\,\partial_y\), \(X_1=\partial_y\); the \(X_0\)
flow \(y(t)=y_0-2x_0 t-t^2\) is polynomial in \(t\): \textbf{complete}.
A polynomial pass baseline paralleling A008.
\end{example}

\section{Toward a dichotomy (open)}
A007 shows the incomplete locus forced by Lemma~\ref{lem:B1} can be as thin
as measure zero for general smooth local diffeomorphisms. A007 is
\emph{not} polynomial. A009 (Pinchuk) supplies the first atlas polynomial
2D non-injective example with \emph{open-set} incompleteness, consistent with
A001's open failures in dimension~3 and with A010 as a polynomial complete
baseline. The candidate claim --- polynomial Keller-type maps never exhibit
A007-style thin incompleteness --- remains open as a \emph{full} theorem, but
is reduced by the restricted lemmas of \S\ref{sec:B5}.

\section{Restricted dichotomy lemmas (new in v0.5)}
\label{sec:B5}
Full proofs: \texttt{PROGRAM-B-B5-poly-dichotomy-lemmas.md}.
CAS: \texttt{cas\_atlas\_B5\_poly\_dichotomy\_B001.json}.

\begin{lemma}[Univariate polynomial local diffeos are global]
\label{lem:B52}
If \(f\in\mathbb{R}[t]\) satisfies \(f'(t)\neq 0\) for all \(t\in\mathbb{R}\), then
\(f:\mathbb{R}\to\mathbb{R}\) is a global diffeomorphism.
\end{lemma}
\begin{proof}
\(f'\) has constant strict sign, so \(f\) is strictly monotone. Even degree is
impossible (equal infinite limits \(\Rightarrow\) not injective). Odd degree gives
opposite infinite limits, hence bijectivity; inverse is \(C^\infty\) by IFT.
\end{proof}

\begin{lemma}[Polynomial product maps have complete duals]
\label{lem:B53}
If \(F(x)=\bigl(f_1(x_1),\ldots,f_n(x_n)\bigr)\) with each \(f_i\in\mathbb{R}[t]\) and
\(\prod_i f_i'\neq 0\), then \(F\) is a global diffeomorphism and every dual field
\(X_i=(1/f_i')\partial_{x_i}\) is complete.
\end{lemma}
\begin{proof}
Apply Lemma~\ref{lem:B52} coordinatewise; dual ODEs integrate by
\(f_i(x_i(t))=f_i(x_i(0))+t\).
\end{proof}

\begin{lemma}[Cylinder criterion]
\label{lem:B51}
If \(\operatorname{int}\pi_j\bigl(\mathbb{R}^n\setminus F(\mathbb{R}^n)\bigr)\neq\emptyset\),
then the incompleteness locus of \(X_j\) contains a nonempty open set.
\end{lemma}
\begin{proof}
On the open set of domain points whose \(F\)-image projects into that interior,
the straight-line orbit in the target meets the complement of the image in finite
time, so the dual lift is incomplete.
\end{proof}

\begin{lemma}[2D fiber saturation and three regimes]
\label{lem:B545}
Let \(F=(P,Q):\mathbb{R}^2\to\mathbb{R}^2\) be polynomial with \(\det DF\neq 0\).
The incompleteness locus \(U_1\) of the dual field with \(\dot Q=1\) is a union of
connected components of levels of \(P\). The set of bad level values is
semi-algebraic in \(\mathbb{R}\), hence a finite union of points and open intervals.
Thus either \(U_1=\emptyset\) (regime E; e.g.\ A010), or \(U_1\) contains a
nonempty open set (regime O; e.g.\ A009), or \(U_1\) is a finite union of real
algebraic curves (regime T, measure zero) --- the only residual thin channel.
\end{lemma}
\begin{proof}[Proof sketch]
No critical points of \(P\); levels are unions of lines; \(Q\) is a chart on each
component with speed \(1\); completeness iff \(Q\) hits all of \(\mathbb{R}\) on that
component. Semi-algebraicity of the bad-value set is Tarski--Seidenberg.
\end{proof}

\begin{lemma}[Jelonek blocks the A007 asymptotic mechanism]
\label{lem:B56}
If \(F:\mathbb{R}^n\to\mathbb{R}^n\) is polynomial and generically finite and the
asymptotic set \(S_F\) is nonempty, then \(\dim S_F=n-1\)~\cite{Jel93,Jel99}.
In particular \(S_F\) cannot be a nonempty finite set of points, so the A007
pattern (single asymptotic value, thin cylindrical projection, null incomplete
locus) is unavailable to polynomials.
\end{lemma}

\noindent\textbf{Restricted package.} Lemmas~\ref{lem:B52}--\ref{lem:B56} eliminate
thin incompleteness for polynomial \emph{product} maps and ban the A007 mechanism
for all polynomial local diffeos. Full exclusion of regime T in
Lemma~\ref{lem:B545} remains open; no atlas example realizes T.

\section{Regime T partial exclusion (new in v0.6)}
\label{sec:B6}
Full proofs: \texttt{PROGRAM-B-B6-regime-T.md}.
CAS: \texttt{cas\_atlas\_B6\_regime\_T\_B001.json}.

\begin{lemma}[Chamber reduction for regime T]
\label{lem:B60}
For polynomial \(F=(P,Q)\) as in Lemma~\ref{lem:B545}, Hardt chambers have constant
\(Q\)-end type. Hence regime T occurs if and only if every open Hardt chamber of
\(P\) is good and \(\operatorname{Bad}_P(F)\) is a nonempty finite subset of the
atypical values of \(P\).
\end{lemma}

\begin{lemma}[Structured exclusions of regime T]
\label{lem:B61-4}
Regime T is impossible for a polynomial local diffeomorphism \(F:\mathbb{R}^2\to\mathbb{R}^2\)
in any of the following classes:
\begin{enumerate}
\item \(F\) proper (then \(F\) is a global diffeomorphism; duals complete);
\item \(F\) injective (then \(F\) is a polynomial automorphism; duals complete);
\item one component has total degree \(1\) (after affine domain change the map is a
global diffeomorphism);
\item product maps as in Lemma~\ref{lem:B53}, or triangular maps
\(F=(f(x),\alpha(x)y+\beta(x))\) with \(f',\alpha\) never zero.
\end{enumerate}
\end{lemma}
\begin{proof}[Proof sketch]
(1) Proper local diffeos \(\mathbb{R}^n\to\mathbb{R}^n\) are coverings, hence global
diffeos. (2) Injective polynomial endomorphisms of \(\mathbb{R}^n\) are automorphisms
(standard; cf.\ van den Essen). (3) Reduce to \(P=x\); then \(Q_y\neq 0\) forces each
univariate \(y\mapsto Q(u,y)\) to be a global diffeo by Lemma~\ref{lem:B52}, yielding
an explicit global inverse. (4) Lemmas~\ref{lem:B53} and B5.7.
\end{proof}

\begin{lemma}[Model atypical submersion is not a coordinate]
\label{lem:B66}
Let \(P_0=x+x^2 y\). Then \(\nabla P_0\neq 0\) everywhere and \(0\) is atypical, but
there is no \(Q\in\mathbb{R}[x,y]\) with \(\det D(P_0,Q)\) a nonzero constant. Moreover
every \(Q\) that is linear in \(y\) has \(\det D(P_0,Q)\) vanishing somewhere; CAS
finds no \(Q\) of total degree \(\leq 6\) with Jacobian equal to any of
\(\pm 1,x^2+1,y^2+1,x^2+y^2+1,(x^2+1)(y^2+1),(x^2+1)^2,(y^2+1)^2\).
\end{lemma}

\noindent\textbf{Partial T package.} Lemmas~\ref{lem:B60}--\ref{lem:B66} exclude T on
structured subclasses and block the standard atypical first-component model at low
degree. The residual \textbf{OPEN-T} asks whether any polynomial local diffeo has
\(\operatorname{Bad}_P\) a nonempty finite subset of \(\operatorname{Atyp}(P)\) with
all Hardt chambers good. No atlas example realizes T; full dichotomy remains a
conjecture.
