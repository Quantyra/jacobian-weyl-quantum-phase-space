\begin{abstract}
Let \(\psi\) be the polynomial Weyl endomorphism of the Alp\"oge--Fable seed
(restated): \(Q=F(q)\), \(P=J^{-T}p\). With the physics normalization
\([q_i,p_j]=i\delta_{ij}\) and \(q_i^*=q_i\), \(p_j^*=p_j\), \(\psi\) is a unital
\(*\)-endomorphism (algebraic SOS positivity). We construct its algebraic
endomorphism correspondence and obstruct: (i)~unital/\allowbreak nondegenerate
\(*\)-endomorphisms of \(C_0(\mathbb{R}^3)\) extending composition with non-proper
\(F\); (ii)~Bogoliubov/quasifree endomorphisms of the CCR algebra (nonlinear
\(F\)). A former ``Free-Strict-Regular-v2'' nondegeneracy obstruction is
\textbf{withdrawn} as incorrect. We then \textbf{construct} a genuine normal
unital \(*\)-homomorphism (Koopman pullback) on the bounded-Borel position
sector, valid for any local-diffeomorphism \(F\) regardless of properness or
surjectivity. For the joint \(q\)--\(p\) sector we record: no choice-free
Joint-Stone-Canonical package at the \(H_1\) leg (A001 deficiency
\((\infty,\infty)\)); under the G4 companion \(H_0,H_2\) deficiency pairs
(not an A001 claim)---no full-triple Joint-Stone package;
\textbf{obstruction} for multiplicative single-generator Joint-Stone-Hom-1
(Theorem~J4); \textbf{obstruction} for mere unital-normal-CP
Joint-Stone-CP-1 (Theorem~J5); and \textbf{obstruction} for the weaker
Unitary-Image-CP-1 pack (Theorem~J6: unitary image of Schr\"odinger Stone
unitaries without any SA-extension-of-\(H_1\) axiom, same geometry).
Bare Joint-Form-Core is constructed on \(C_c^\infty\); Joint-Form with ESS
uniqueness fails for \(H_1\). Non-Stone CP of the position pullback
\emph{splits}: Diag-CP-\(\Phi_0\) (\(\Phi_0\circ E\) on the joint vNa)
\textbf{construct}. Full-\(\psi\) abstract \(C^*\) without Hom/unitary-image
\emph{splits} further (Theorem~J7): Full-$\psi$-BT-Envelope
\textbf{construct} (concrete \(C^*\) from \(\Phi_0(C_b)\) and bounded
transforms of minimal dual momenta); Full-$\psi$-CFC-SA-1
\textbf{obstruction} (SA continuous functional calculus / unitary Cayley
image reduces to J6); Full-$\psi$-CP-Weyl-\(C^*\) remains open (narrow).
No dual \(F\)-translations, gates, or advantage. Companion
to~\cite{A001}. Pack: \texttt{COMPANION-PACK.md}.
\end{abstract}

\section{Domain}
\begin{definition}
\(\mathcal{W}\): polynomial Weyl algebra over \(\mathbb{C}\) with
\([q_i,p_j]=i\delta_{ij}\), \([q_i,q_j]=[p_i,p_j]=0\), and involution
\(q_i^*=q_i\), \(p_j^*=p_j\). (Consistent: \([q,p]^*=-[q,p]\) and
\((i\delta)^*=-i\delta\).) With real \(B=J^{-T}\),
\[
\psi(q_i)=F_i(q),\qquad
\psi(p_j)=\sum_k B_{jk}(q)\,p_k
\]
(coefficients left of momenta) extends to a unital algebra endomorphism.
\end{definition}

\begin{remark}[Normalization]
Earlier drafts used \([q,p]=\delta\) with self-adjoint generators, which is
inconsistent under \(*\). This note uses the standard physics normalization
\([q,p]=i\).
\end{remark}

\begin{theorem}[Algebraic \(*\)-SOS]
Assume \(F,B\) real and the Piola/\allowbreak dual-field identity \(\operatorname{div} B=0\)
(row divergence free; true for the A001 cotangent lift). Then each
Schr\"odinger operator \(H_j=-iX_j\) is \emph{symmetric} on \(C_c^\infty(\mathbb{R}^3)\)
(not claimed essentially self-adjoint---cf.~\cite{A001} for \(H_1\)).
At the algebraic level the formal adjoint correction
\(-i\sum_k\partial_k B_{jk}\) vanishes, so
\[
P_j^* = P_j
\quad\text{in }\mathcal{W}
\]
with the physics involution. Hence \(\psi\) is a unital \(*\)-endomorphism and
\(\mathrm{id}_n\otimes\psi\) preserves hermitian-square cones.
\end{theorem}

\begin{definition}[Algebraic correspondence]
Let \(E=\mathcal{W}\) as a \emph{right} \(\mathcal{W}\)-module by \(\xi\cdot b=\xi b\),
with \emph{left} action \(a\cdot\xi=\psi(a)\xi\) and
\(\langle\xi,\eta\rangle_{\mathcal{W}}=\xi^*\eta\).
\end{definition}

\section{Obstructions (retained)}
\begin{theorem}[C0 composition endo]
\label{thm:c0}
There is no \(*\)-homomorphism \(\alpha:C_0(\mathbb{R}^3)\to C_0(\mathbb{R}^3)\)
(nondegenerate/\allowbreak approximately unital in the usual \(C_0\) sense) with
\(\alpha(f)=f\circ F\) for all \(f\in C_0(\mathbb{R}^3)\), because \(F\) is not proper
so \(f\circ F\notin C_0\) for some \(f\in C_0\).
\end{theorem}

\begin{theorem}[Bogoliubov / quasifree]
No Bogoliubov (linear quasifree) endomorphism of the CCR \(C^*\)-algebra---i.e.\ a
map induced by a linear symplectic contraction on the one-particle space,
acting on Weyl unitaries by \(W(z)\mapsto W(Tz)\)---can reproduce the Heisenberg
picture dynamics of \(\psi\) on the Weyl one-parameter groups associated with the
field coordinates in a regular representation: those Bogoliubov maps act
linearly on one-particle fields, while \(q\mapsto F(q)\) is a nonlinear
polynomial map (total degree \(>1\); A001 seed has terms up to total degree \(7\)).
\end{theorem}

\begin{remark}[Not claimed]
Polynomial \(q,p\) are not themselves elements of the CCR \(C^*\)-algebra; the
comparison is through regular representations / affiliated fields / Weyl
groups. We do \emph{not} claim every continuous unital \(*\)-endo of the CCR
algebra fails---only Bogoliubov/quasifree. We do \emph{not} claim
\(f\mapsto M_{f\circ F}\) fails as a map \(C_b\to B(L^2)\). Arveson extensions of
the abelian composition map need not implement the momentum part of \(\psi\).
\end{remark}

\section{Construct: Koopman position sector}
\begin{theorem}[Koopman pullback]
Let \(\mathcal{B}_b(\mathbb{R}^3)\) be the abelian von Neumann algebra of
bounded Borel functions of \(Q\). For \(F\) a local diffeomorphism
(\(\det DF\neq 0\) everywhere, in particular the A001 seed), the pullback
\[
\Phi_0(f):=f\circ F,
\qquad f\in\mathcal{B}_b(\mathbb{R}^3),
\]
is a well-defined \textbf{unital normal \(*\)-homomorphism}
\(\mathcal{B}_b(\mathbb{R}^3)\to\mathcal{B}_b(\mathbb{R}^3)\) (hence a normal
unital CP map), for \emph{any} such \(F\)---regardless of properness or
surjectivity.
\end{theorem}
\begin{proof}
\(F\) continuous \(\Rightarrow f\circ F\) Borel and bounded whenever \(f\) is.
Local diffeomorphism \(\Rightarrow\) \(F\) is Lebesgue-nonsingular (preimages
of null sets are null, by the inverse function theorem on a countable cover
of charts), so \(\Phi_0\) descends to a.e.-equivalence classes.
\((f\circ F)^*=\overline{f\circ F}\) and \((fg)\circ F=(f\circ F)(g\circ F)\)
pointwise, hence a.e. A unital \(*\)-homomorphism of von Neumann algebras is
automatically normal and completely positive.
\end{proof}

\begin{remark}
This resolves, at the position-sector level, the intuition behind the
withdrawn Regular-v2: non-surjectivity of \(F\) does \emph{not} obstruct the
bounded-Borel Koopman pullback. The C0 obstruction (Theorem~\ref{thm:c0}) is
specific to the \emph{vanishing-at-infinity} algebra \(C_0\), not to bounded
Borel functions.
\end{remark}

\section{Joint \(q\)--\(p\) extension: eight-slice status}
\(\Phi_0\) only captures \(\psi\) on the position generators. The joint
sector splits as follows (full write-up:
\texttt{PROGRAM-C-Free-Strict-Abstract-Joint.md};
J4: \texttt{PROGRAM-C-residual-J4-Joint-Stone-Hom.md};
J5: \texttt{PROGRAM-C-residual-mere-CP-Joint-Stone.md};
J6: \texttt{PROGRAM-C-residual-CP-without-Stone.md};
J7: \texttt{PROGRAM-C-residual-abstract-Cstar-full-psi.md}).

\paragraph{(i) Position sector --- CONSTRUCT.}
The Koopman pullback \(\Phi_0\) is a normal unital \(*\)-homomorphism on
bounded Borel functions of \(Q\) (previous section).

\paragraph{(ii) Joint-Stone-Canonical --- CONDITIONAL-OBSTRUCTION.}
A \emph{Joint-Stone} package asks for self-adjoint extensions
\(\widetilde H_j\supset H_j=-iX_j\) together with a unital normal CP map
\(\Phi\) extending \(\Phi_0\) and implementing
\(\Phi(e^{is P_j^{\mathrm{Sch}}})=e^{is\widetilde H_j}\).
\emph{Joint-Stone-Canonical} further requires the package to be uniquely
fixed by the algebraic data \((\psi,F,B)\) alone.
\begin{itemize}
\item \textbf{Theorem J2} (A001 \(H_1\) only): since
\((n_+,n_-)=(\infty,\infty)\) for \(H_1=-iX_1\)~\cite{A001}, von Neumann's
theorem yields a continuum of inequivalent SA extensions; \((\psi,F,B)\)
does not select one. Hence no Joint-Stone-Canonical package exists.
\item \textbf{Theorem J3} (G4 companion, not A001): the companion note
\texttt{G4-H0-H2-deficiency-bounds.md} records
\((n_+,n_-)=(\infty,0)\) for \(H_0\) and \((0,\infty)\) for \(H_2\), so
those minimal operators admit \emph{no} SA extensions. Under those pairs,
no full-triple Joint-Stone package \((\Phi;\widetilde H_0,\widetilde H_1,\widetilde H_2)\)
exists at all.
\end{itemize}

\paragraph{(iii) Joint-Stone-Hom-1 --- OBSTRUCTION.}
A \emph{Joint-Stone-Hom-1} package asks for a self-adjoint extension
\(\widetilde H_1\supset H_1\) and a unital normal \(*\)-homomorphism
\(\Phi\) on the vNa generated by bounded Borel functions of \(Q\) and
\(\{e^{is P_1^{\mathrm{Sch}}}\}\) with \(\Phi|_{\mathcal{B}_b(Q)}=\Phi_0\) and
\(\Phi(e^{is P_1^{\mathrm{Sch}}})=e^{is\widetilde H_1}\).
\begin{theorem}[J4]
No Joint-Stone-Hom-1 package exists for the A001 seed.
\end{theorem}
\begin{proof}[Proof sketch]
Multiplicativity yields dual-\(F_1\) Heisenberg covariance
\(e^{-is\widetilde H_1}(f\circ F)e^{is\widetilde H_1}=f(F_0,F_1+s,F_2)\).
Hence \((F_1,\widetilde H_1)\) is a regular Schr\"odinger couple and
\(F_0,F_2\) act on the multiplicity space only. The joint spectral measure of
\((F_0,F_1,F_2)\) must then have \(s\)-independent Lebesgue density. But
\(|\det DF|\equiv 2\), so that density equals \(m/2\) with
\(m=\#F^{-1}\). For A001, \(m\) takes the values \(3\) and \(1\) on two
positive-measure open slabs near \((a,c)=(0,2)\). Contradiction.
Full write-up: \texttt{PROGRAM-C-residual-J4-Joint-Stone-Hom.md}.
\end{proof}

\paragraph{(iv) Joint-Stone-CP-1 --- OBSTRUCTION.}
A \emph{Joint-Stone-CP-1} package weakens Hom to mere unital normal CP:
self-adjoint extension \(\widetilde H_1\supset H_1\), unital normal CP map
\(\Phi\) on the same vNa, \(\Phi|_{\mathcal{B}_b(Q)}=\Phi_0\), and
\(\Phi(e^{is P_1^{\mathrm{Sch}}})=e^{is\widetilde H_1}\).
\begin{theorem}[J5]
No Joint-Stone-CP-1 package exists for the A001 seed.
\end{theorem}
\begin{proof}[Proof sketch]
Unital CP and Kadison--Schwarz: if \(U(s)=e^{is P_1^{\mathrm{Sch}}}\) and
\(\Phi(U(s))\) is unitary, then \(U(s)\) lies in the multiplicative domain
\(\mathcal{D}_\Phi\), so
\(\Phi(U(s)aU(s)^*)=\Phi(U(s))\Phi(a)\Phi(U(s))^*\) for all \(a\).
Taking \(a=f(Q)\) recovers the same dual-\(F_1\) Heisenberg covariance as in
J4. The SvN product structure and \(s\)-dependent A001 sheet count then yield
the same contradiction. Full write-up:
\texttt{PROGRAM-C-residual-mere-CP-Joint-Stone.md}.
\end{proof}

\paragraph{(v) Unitary-Image-CP-1 --- OBSTRUCTION (new); Diag-CP-\(\Phi_0\) --- CONSTRUCT.}
Drop the SA-extension-of-\(H_1\) axiom entirely. Let \(M\) be the vNa generated
by \(\mathcal{B}_b(Q)\) and \(\{U(s)=e^{is P_1^{\mathrm{Sch}}}\}\).
\begin{itemize}
\item \textbf{Unitary-Image-CP-1:} unital normal CP \(\Phi:M\to B(H)\) with
\(\Phi|_{\mathcal{B}_b(Q)}=\Phi_0\) and \(\Phi(U(s))\) unitary for all \(s\)
(no \(\widetilde H_1\supset H_1\) required).
\item \textbf{Diag-CP-\(\Phi_0\):} \(\Phi_{\mathrm{diag}}=\Phi_0\circ E\) with
\(E:M\to\mathcal{B}_b(Q)\) the normal conditional expectation onto positions
(crossed-product / injectivity).
\end{itemize}
\begin{theorem}[J6]
No Unitary-Image-CP-1 package exists for the A001 seed.
\end{theorem}
\begin{proof}[Proof sketch]
Unitary image \(\Rightarrow U(s)\in\mathcal{D}_\Phi\) \(\Rightarrow\) same
dual-\(F_1\) Heisenberg covariance as J4/J5 \(\Rightarrow\) same SvN /
sheet-count contradiction. The dual-field SA link is inessential.
\end{proof}
\begin{theorem}[J6-C]
\(\Phi_{\mathrm{diag}}=\Phi_0\circ E\) is a unital normal CP map on \(M\)
extending \(\Phi_0\), with \(\Phi_{\mathrm{diag}}(U(s))=0\) for \(s\neq 0\)
(hence no Stone/SA momentum implementation). Schr\"odinger--Heisenberg
conjugates of position observables pull back by \(\Phi_0\) automatically.
\end{theorem}
Full write-up: \texttt{PROGRAM-C-residual-CP-without-Stone.md}.

\paragraph{(vi) Joint-Form split.}
\begin{itemize}
\item \textbf{Joint-Form-Core --- CONSTRUCT:} sesquilinear forms of the
minimal dual momenta \(H_j=-iX_j\) on \(C_c^\infty\) realize \(\psi(p_j)\)
together with \(\Phi_0\) on positions (no SA/Stone requirement).
\item \textbf{Joint-Form-ESS-1 --- OBSTRUCT:} requiring ESS uniqueness for

\paragraph{(vii) Full-\(\psi\) abstract \(C^*\) --- SPLIT (J7).}
Three named packs (full write-up:
\texttt{PROGRAM-C-residual-abstract-Cstar-full-psi.md}):
\begin{itemize}
\item \textbf{Full-$\psi$-BT-Envelope:} concrete
\(C^*\)-algebra
\(\mathfrak{A}_\psi\subset B(H)\) generated by
\(\{(f\circ F)(Q):f\in C_b(\mathbb{R}^3)\}\) and the bounded transforms
\(Z_j=Z(\overline{H_j})\) of the closed minimal dual momenta
(\(Z(A)=A(I+A^*A)^{-1/2}\)).
\item \textbf{Full-$\psi$-CFC-SA-1:} unital normal CP extending \(\Phi_0\)
that recovers SA continuous functional calculus of some
\(\widetilde A_1\) for \(P_1^{\mathrm{Sch}}\) (equivalently: unitary
Cayley image of \(C(P_1^{\mathrm{Sch}})\)).
\item \textbf{Full-$\psi$-CP-Weyl-\(C^*\):} unital CP from a \(C^*\)
completion of the polynomial Weyl algebra extending algebraic \(\psi\),
without SA-CFC/unitary-image axioms.
\end{itemize}
\begin{theorem}[J7-C]
Full-$\psi$-BT-Envelope is a well-defined unital \(C^*\)-subalgebra of
\(B(H)\) carrying position content of \(\psi\) via \(\Phi_0(C_b)\) and
momentum-form content via \(\{Z_j\}\). For \(j=1\), \(Z_1\) is symmetric
but not self-adjoint (A001: \(H_1\) not ESS). No SA-extension choice, no
unitary Stone image, no \(*\)-hom of full CCR is used.
\end{theorem}
\begin{theorem}[J7]
No Full-$\psi$-CFC-SA-1 package exists for the A001 seed.
\end{theorem}
\begin{proof}[Proof sketch]
Unitary Cayley image places \(C(P_1^{\mathrm{Sch}})\) in the multiplicative
domain; continuous functional calculus recovers unitary Stone images
\(\Phi(e^{is P_1^{\mathrm{Sch}}})\), yielding Unitary-Image-CP-1, which
Theorem~J6 obstructs.
\end{proof}
Full-$\psi$-CP-Weyl-\(C^*\) remains \textbf{open} (J7-O; narrowed residual).
Diag-CP-\(\Phi_0\) is \emph{not} full-\(\psi\).

\section{Non-claims}
No channel on \(B(H)\) implementing the full \(\psi\).
No gates/advantage. No dual \(F\)-translations. No ``all quantization fails.''
No claim that every joint CP construction is impossible---only
Joint-Stone-Canonical (J2), full-triple Joint-Stone under G4 pairs (J3),
multiplicative Joint-Stone-Hom-1 (J4), mere-CP Joint-Stone-CP-1 (J5),
Unitary-Image-CP-1 (J6), Full-$\psi$-CFC-SA-1 (J7), and Joint-Form-ESS-1.
Diag-CP-\(\Phi_0\) is not a momentum channel and is not full-\(\psi\).
Full-$\psi$-BT-Envelope is a concrete \(C^*\) envelope, not a channel.
Full-$\psi$-CP-Weyl-\(C^*\) without SA-CFC is not claimed impossible.
No promotion of G4 \(H_0/H_2\) pairs to A001 theorems.
