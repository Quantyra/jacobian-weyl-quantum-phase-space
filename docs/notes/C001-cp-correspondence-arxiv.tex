\begin{abstract}
Let \(\psi\) be the polynomial Weyl-algebra endomorphism induced by the
Alp\"oge--Fable Keller seed (restated) via \(Q=F(q)\) and \(P=J^{-T}p\).
We prove \(\psi\) is a unital \(*\)-endomorphism (algebraic SOS positivity),
construct its algebraic endomorphism correspondence, and obstruct two standard
continuous packages: unital \(*\)-endomorphisms of \(C_0(\mathbb{R}^3)\) extending
composition with \(F\), and Bogoliubov endomorphisms of the CCR \(C^*\)-algebra.
We also obstruct unital CP maps that are \(C_0\)-composition compatible.
A free (non-C0, non-Bogoliubov) unital CP extension to some \(C^*\) completion
remains open. No dual \(F\)-translations, gates, or advantage are claimed.
Companion to the A001 deficiency note~\cite{A001}.
\end{abstract}

\section{Domain}
\begin{definition}
\(\mathcal{W}\): polynomial Weyl algebra over \(\mathbb{C}\), \([q_i,p_j]=\delta_{ij}\),
involution \(q_i^*=q_i\), \(p_j^*=p_j\). With \(B=J^{-T}\),
\[
\psi(q_i)=F_i(q),\qquad
\psi(p_j)=\sum_k B_{jk}(q)\,p_k
\]
extends to a unital algebra endomorphism \(\psi:\mathcal{W}\to\mathcal{W}\).
\end{definition}

\begin{theorem}[Algebraic \(*\)-SOS]
If \(F,B\) are real, then \(\psi\) is a unital \(*\)-endomorphism, hence preserves
hermitian-square cones at all matrix levels \(\mathrm{id}_n\otimes\psi\).
\end{theorem}

\section{Non-properness}
\begin{theorem}
\(F:\mathbb{R}^3\to\mathbb{R}^3\) is not proper: along the explicit incomplete
\(X_1\)-orbit \(\gamma(t)\) of~\cite{A001}, \(F(\gamma(t))=(0,t,2)\) stays bounded
while \(\|\gamma(t)\|\to\infty\) (CAS \texttt{cas\_F\_not\_proper\_C1c\_A001.json}).
\end{theorem}

\section{Obstructions}
\begin{theorem}[C0 \(*\)-endo]
No unital \(*\)-endomorphism of \(C_0(\mathbb{R}^3)\) extends \(f\mapsto f\circ F\)
on polynomials.
\end{theorem}

\begin{theorem}[Bogoliubov]
No Bogoliubov endomorphism of the CCR \(C^*\)-algebra restricts to \(\psi\) on
\(\mathcal{W}\) (\(F\) is nonlinear).
\end{theorem}

\begin{theorem}[C0-compatible CP]
No unital CP map on a \(C^*\) algebra containing \(C_0(\mathbb{R}^3)\) as position
functions can extend \(\psi\) while agreeing with composition on a dense
subalgebra of that \(C_0\).
\end{theorem}

\section{Construct: algebraic correspondence}
\begin{definition}[Endomorphism correspondence]
Let \(E=\mathcal{W}\) as a right \(\mathcal{W}\)-module, left action
\(a\cdot\xi=\psi(a)\xi\), and \(\langle\xi,\eta\rangle=\xi^*\eta\).
\end{definition}
This is the standard algebraic Hilbert-module correspondence implementing the
unital \(*\)-endo \(\psi\). It does \emph{not} by itself yield a continuous CP map
on a \(C^*\) completion.

\section{Open}
Free unital CP extension of \(\psi\) to some \(C^*\) completion of \(\mathcal{W}\)
without C0-composition or Bogoliubov axioms: \textbf{not constructed, not ruled out}.

\section{Non-claims}
No channel on \(B(H)\). No gates/advantage. No dual \(F\)-translations.
No claim every CP extension is impossible in every axiomatics.
Algebraic correspondence \(\neq\) physical instrument.
