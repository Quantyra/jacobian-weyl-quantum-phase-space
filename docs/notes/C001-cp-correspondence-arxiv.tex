\begin{abstract}
Let \(\psi\) be the polynomial Weyl endomorphism of the Alp\"oge--Fable seed
(restated): \(Q=F(q)\), \(P=J^{-T}p\). We record that \(\psi\) is a unital
\(*\)-endomorphism (algebraic SOS positivity) and construct its algebraic
endomorphism correspondence. We obstruct standard packages (C0 \(*\)-endo,
Bogoliubov, CCR \(*\)-hom, C0-compatible CP) and a \textbf{Free-Strict-Regular}
package: unital CP on \(B(L^2)\) with spectral implementation
\(\Phi(f(Q))=M_{f\circ F}\) for \(f\in C_b\) and nondegeneracy on \(C_0(Q)\).
A fully abstract C\(^*\) envelope CP map with no spatial axioms remains open.
No dual \(F\)-translations, gates, or advantage. Companion to~\cite{A001}.
\end{abstract}

\section{Domain and algebraic positivity}
\begin{definition}
\(\mathcal{W}\): polynomial Weyl algebra, \([q_i,p_j]=\delta_{ij}\), \(q_i^*=q_i\),
\(p_j^*=p_j\). With \(B=J^{-T}\),
\(\psi(q_i)=F_i(q)\), \(\psi(p_j)=\sum_k B_{jk}(q)p_k\).
\end{definition}

\begin{theorem}[Algebraic \(*\)-SOS]
Real \(F,B\) \(\Rightarrow\) \(\psi\) unital \(*\)-endo \(\Rightarrow\) SOS cones preserved
at all matrix levels.
\end{theorem}

\begin{definition}[Algebraic correspondence]
\(E=\mathcal{W}\) as right \(\mathcal{W}\)-module, \(a\cdot\xi=\psi(a)\xi\),
\(\langle\xi,\eta\rangle=\xi^*\eta\).
\end{definition}

\section{Free-Strict-Regular NO-GO}
\begin{theorem}[Free-Strict-Regular-v2]
There is no unital CP map \(\Phi:B(L^2(\mathbb{R}^3))\to B(L^2(\mathbb{R}^3))\) such that
\begin{enumerate}
\item \(\Phi(f(Q))=M_{f\circ F}\) for all \(f\in C_b(\mathbb{R}^3)\);
\item whenever \(P\in B(L^2)\) satisfies \(\Phi(f(Q))P=0\) for all \(f\in C_0(\mathbb{R}^3)\),
then \(P=0\) (nondegeneracy on positions).
\end{enumerate}
\end{theorem}
\begin{proof}
Let \(P=M_{1_{\mathbb{R}^3\setminus\overline{F(\mathbb{R}^3)}}}\). Non-surjectivity gives \(P\ne 0\).
For \(f\in C_0(\mathbb{R}^3)\), \(f\circ F=0\) on \(\operatorname{supp}P\), hence
\(M_{f\circ F}P=0\), i.e.\ \(\Phi(f(Q))P=0\), contradicting nondegeneracy.
\end{proof}

\begin{remark}
Stinespring data for this pack are vacuous (empty domain).
\end{remark}

\section{Free-Strict-Abstract (residual OPEN)}
Unital CP on some abstract C\(^*\) envelope of \(\mathcal{W}\) with
\textbf{no} spectral-position, C0-composition, Bogoliubov, or CCR-\(*\)-hom axiom:
not constructed beyond the algebraic correspondence and finite-level algebraic
Stinespring; not ruled out.

\section{Non-claims}
No channel on \(B(H)\) constructed. No gates/advantage. No dual \(F\)-translations.
No claim every abstract CP extension fails.
