\begin{abstract}
We study the Alp\"oge--Fable Keller map $F$ ($\det DF\equiv-2$, three-point collision). Novelty is Poisson/Weyl lifts and the dual transport operator $H=-iX_1$ on $C_c^\infty(\mathbb{R}^3)$, which has deficiency indices $(n_+,n_-)=(\infty,\infty)$: not essentially self-adjoint, infinitely many self-adjoint extensions, none preferred by algebraic CCR data. No gates, channels, or computational advantage. Seed discovery not claimed.
\end{abstract}

F_2&=2x - 3x^2 y - x^3 z.
\end{align*}

\section{Main theorems}
\begin{theorem}[A]
$\det DF\equiv-2$; three-point collision; $F$ noninjective. (Alp\"oge--Fable seed restated.)
\end{theorem}
\begin{theorem}[B--C]
Cotangent lift $\Phi$ is polynomial Poisson and noninjective. Dual fields commute and are divergence-free (Piola); $H_j=-iX_j$ symmetric on $C_c^\infty$.
\end{theorem}
\begin{theorem}[D]
$X_1$ incomplete: explicit curve through $(1,0,0)$ escapes as $t\to\tfrac12^-$, with $F(\gamma(t))=(0,t,2)$.
\end{theorem}
\begin{theorem}[F]
In coordinates $(a,s,c)=F(q)$, $X_1=\partial_s$ and $dq=\tfrac12\,da\,ds\,dc$. Open forward wall near $(a,c)=(0,2)$ and backward wall near $(1/54,2)$ yield $L^2$ deficiency functions $u_\pm$, hence $(n_+,n_-)=(\infty,\infty)$.
\end{theorem}
\begin{theorem}[E]
$H$ is not essentially self-adjoint on $C_c^\infty(\mathbb{R}^3)$.
\end{theorem}
\begin{corollary}
Infinitely many self-adjoint extensions exist; algebraic data select none.
\end{corollary}

\section{Non-claims}
No unitary gate, channel, CP map, or computational advantage. No discovery claim for $F$. No preferred physical extension.
